#### 1 补最小的括号使得[]()匹配
```python3
# ss = input().strip()
ss = '([]([](])))([]()()]([][[)[()[)]([[(])][][[[([)]'
stack = []
ans = 0
for s in ss:
    if not stack and (s == ']' or s == ' )'):
        ans += 1
    elif not stack:
        stack.append(s)
    else:
        top = stack[-1]
        if s == '(' or s == '[':
            stack.append(s)
        if (s == ')' and top == '(' ) or (s == ']' and top == '['):
            stack.pop(-1)
        elif s == ')' or s == ']':
            ans+=1

    print(stack)

print(ans,len(stack))

```

#### 2 ax^2+x+b = 0 与 x=c x=d 围成的面积
```python3
T = int(input().strip())


def fun(A, B, C, D):
    A, B, C, D = float(A), float(B), float(C), float(D)
    x = C
    y1 = A / 3 * (x ** 3) + (x ** 2) / 2 + B * x
    x = D
    y2 = A / 3 * (x ** 3) + (x ** 2) / 2 + B * x
    return round(y2 - y1, 6)


ans = []
for i in range(T):
    A, B, C, D = list(map(int, input().strip().split()))
    ans.append(str(fun(A, B, C, D)))

print('\n'.join(ans))
```
